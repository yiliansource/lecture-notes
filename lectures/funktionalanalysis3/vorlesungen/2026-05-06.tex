
Haben wir $C^*$-Algebren $\Aa \leq \Ba$ mit respektiven Abbildungen $\varepsilon_\Aa, \varepsilon_\Ba$, so können wir schreiben
\[
    \varepsilon_{\Ba}^{-1}(b) = \sum_{n=1}^\infty b^n.
\]
Damit ist also
\[
    \restr{\varepsilon_\Ba^{-1}}{\Aa_+ \cap B_1^\Aa (0)} = \varepsilon_\Aa^{-1},
\]
womit folgt
\[
    \varepsilon_\Aa = \restr{\varepsilon_\Ba}{\Aa_+ \cap B_1^\Aa (0)}.
\]

\begin{lemma}
    Sei $\Aa$ eine $C^*$-Algebra, so ist das Netz $(\varepsilon(q))_{q \in \Aa_+}$ (mit Werten in $\Aa_+ \cap B_1^\Aa (0)$) eine approximative Eins, das heißt
    \[
        \lim_{q \in \Aa_+} \varepsilon(q) a = \lim_{q \in \Aa_+} a \varepsilon(q) = a, \quad \forall a \in \Aa.
    \]
\end{lemma}

\begin{proof}
    Wir erinnern, dass wir für Elemente $p, q \in \Aa_+$ bereits gezeigt haben:
    \begin{itemize}
        \item $p \leq q \implies \Vert p \Vert \leq \Vert q \Vert$
        \item $p \leq \Vert p \Vert e$
        \item $p \leq e \Leftrightarrow \Vert p \Vert \leq 1$
        \item Für $c \in \Aa$ und $p \leq q$ gilt $c^* p c \leq c^* q c$.
    \end{itemize}
    Es reicht nun die Aussage für $a \in \Aa_+$ zu zeigen, da wir ein allgemeines $a \in \Aa$ in Real- und Imaginärteil zerlegen können (welche selbstadjungiert sind), und anschließend die selbstadjungierten Elemente in Differenz zweier positiver Elemente. Vermöge einer Streckung können wir sogar $\Vert a \Vert < 1$ annehmen.

    Sei nun $\eta > 0$ und setze $q_\eta \coloneqq \frac{1}{\eta^2} a \in \Aa_+$. Für beliebiges $q \in \Aa_+, q \geq q_\eta$ gilt nun $\varepsilon(q_\eta) \leq \varepsilon(q) \leq e$ und $e - \varepsilon(q) \leq e$. Damit folgt
    \begin{align*}
        0
        &\leq
        (e - \varepsilon(q))^2
        =
        \sqrt{e - \varepsilon(q)} (e - \varepsilon(q)) \sqrt{e - \varepsilon(q)}
        \leq
        e - \varepsilon(q) \\
        &\leq
        e - \varepsilon(q_\eta)
        = \left( e + \frac{1}{\eta^2} a \right)^{-1}.
    \end{align*}
    Konjugieren mit $a$ liefert
    \begin{align*}
        0
        &\leq
        a(e - \varepsilon(q))^2 a
        \leq
        \sqrt{a} \left[ \sqrt{a} \left( e + \frac{1}{\eta^2} a \right)^{-1} \sqrt{a} \right] \sqrt{a}.
    \end{align*}
    Der Term in der $[\cdots]$ Klammer ist gleich
    \[
        a \left( e + \frac{1}{\eta^2} a \right)^{-1} = \eta^2 \frac{1}{\eta^2} a \left( e + \frac{1}{\eta^2} a \right)^{-1} = \eta^2 \varepsilon(q_\eta) \leq \eta^2 e,
    \]
    womit insgesamt die Ungleichung
    \[
        0
        \leq
        a (e - \varepsilon(q))^2 a \leq \eta^2 a
    \]
    folgt. Anwenden der Norm liefert
    \[
        \Vert a (e - \varepsilon(q))^2 a \Vert \leq \eta^2 \Vert a \Vert < \eta^2.
    \]
    Damit folgt nun
    \begin{align*}
        \Vert a - \varepsilon(q) a \Vert^2
        &=
        \Vert a - a \varepsilon(q) \Vert^2
        =
        \Vert a (e - \varepsilon(q) ) \Vert^2
        =
        \Vert a (e - \varepsilon(q))^2 a \Vert
        <
        \eta^2.
    \end{align*}
    Damit folgt die erste Gleichheit in der Aussage, und aus Symmetriegründen auch der Rest.
\end{proof}

\begin{proposition}
    Sei $\Aa$ eine $C^*$-Algebra und $J$ ein echtes, abgeschlossenes Ideal von $\Aa$. Dann gilt:
    \begin{enumerate}
        \item $J^* = J$. Insbesondere ist $J$ eine $C^*$-Unteralgebra.
        \item Die Faktorraumnorm auf $A/_J$ erfüllt
        $$ \Vert [a] \Vert = \lim_{q \in J_+} \Vert a - a \varepsilon(q) \Vert. $$
        \item $A/_J$ versehen mit der Faktorraumnorm ist eine $C^*$-Algebra, falls $[a]^* \coloneqq [a^*]$.
    \end{enumerate}
\end{proposition}

\begin{proof}{\ }
    \begin{enumerate}
        \item Man verifiziert sofort, dass $J$ eine Banach-Unteralgebra ist, es reicht also $J^* \subseteq J$ zu zeigen. Sei $d \in J$ und betrachte die (in $\Aa$) erzeugte $C^*$-Algebra
        $$ \Ca \coloneqq [{d^* d}] = \clspn \{ (d^* d)^n : n \in \N \} \leq J. $$
        Für $q \in \Ca_+$ gilt $\varepsilon(q) \in \Ca_+$ und
        \begin{align*}
            \Vert \varepsilon(q) d^* - d^* \Vert^2
            &=
            \Vert d \varepsilon(q) - d \Vert^2
            =
            \Vert d (\varepsilon(q) - e) \Vert^2
            =
            \Vert (\varepsilon(q) - e) d^* d (\varepsilon(q) - e) \Vert \\
            &\leq
            \Vert \varepsilon(q) - e \Vert \, \Vert d^* d \varepsilon(q) - d^* d \Vert
            \leq
            2 \Vert d^* d \varepsilon(q) - d^* d \Vert.
        \end{align*}
        Mit der approximativen Einheit erhalten wir als Grenzwert für $q \in \Ca_+$, dass der obige Ausdruck gegen $0$ konvergiert. Also ist $d^* \in \cl{J} = J$.

        \item Sei $a \in \Aa, d \in J, q \in J_+$, dann ist $a \varepsilon(q) \in J$ und $0 \leq e - \varepsilon(q) \leq e$. Jetzt gilt
        \begin{align*}
            \Vert [a] \Vert
            &=
            \inf_{j \in J} \Vert a - j \Vert
            =
            \Vert [a - a \varepsilon(q) ] \Vert
            \leq 
            \Vert a - a \varepsilon(q) \Vert \\
            &\leq
            \Vert (a - d)(e - \varepsilon(q)) \Vert + \Vert d - d \varepsilon(q) \Vert
            \leq
            \Vert a - d \Vert + \Vert d - d \varepsilon(q) \Vert.
        \end{align*}
        Nehmen wir den Limes superior, so erhalten wir
        \begin{align*}
            \Vert [a] \Vert
            &\leq
            \limsup_{q \in J_+} \Vert a - a \varepsilon(q) \Vert
            \leq
            \Vert a - d \Vert, \quad \forall a \in \Aa, d \in J.
        \end{align*}
        Anwenden des Infimums über $d \in J$ liefert
        \[
            \Vert [a] \Vert
            \leq
            \limsup_{q \in J_+} \Vert a - a \varepsilon(q) \Vert
            \leq
            \Vert [a] \Vert
            \leq
            \liminf_{q \in J_+} \Vert a - a \varepsilon(q) \Vert
        \]
        Insbesondere existiert der Limes und stimmt mit der Faktorraumnorm überein.

        \item Zunächst gilt
        \begin{align*}
            \Vert (e - \varepsilon(q)) a^* a (e - \varepsilon(q)) \Vert = \Vert a - a \varepsilon(q) \Vert^2,
        \end{align*}
        wobei wegen dem vorigen Punkt die Limiten über $q \in J_+$ existieren und mit $\Vert [a] \Vert^2$ übereinstimmen. Damit gilt
        \begin{align*}
            \Vert [a]^* [a] \Vert
            &=
            \Vert [a^* a] \Vert
            =
            \lim_{q \in J_+} \Vert a^* a - a^* a \varepsilon(q) \Vert
            =
            \lim_{q \in J_+} \Vert a^* a (e - \varepsilon(q)) \Vert \\
            &\geq
            \lim_{q \in J_+} \Vert (e - \varepsilon(q)) a^* a (e - \varepsilon(q)) \Vert
            =
            \Vert [a] \Vert^2. \qedhere
        \end{align*}
    \end{enumerate}
\end{proof}

\begin{example}
    Sei $H$ ein unendlichdimensionaler Hilbertraum und betrachte die kompakten Operatoren $K(H)$ als echtes Unterideal von $\Aa \coloneqq L_b(H)$. Nach dem oben Gezeigten ist $L_b(H)/_{K(H)}$ eine $C^*$-Algebra.
\end{example}

\begin{corollary}
    Seien $\Aa, \Ba$ $C^*$-Algebren und sei $\varphi$ ein $\ast$-Algebrenhomomorphismus von $\Aa$ nach $\Ba$, $\varphi \neq 0$. Dann bildet $\varphi(\Aa)$ eine Unter-$C^*$-Algebra von $\Ba$ und
    \[
        \varphi/_{\ker \varphi} : \Aa/_{\ker \varphi} \to \varphi(\Aa)
    \]
    ist ein isometrischer $\ast$-Algebrenisomorphismus.
\end{corollary}

\begin{proof}
    Zunächst ist $\varphi$ stetig (sogar eine Kontraktion), wie wir bereits gezeigt haben. Weiters ist $\ker \varphi$ ein echtes, abgeschlossenes Ideal in $\Aa$. Nach der vorigen Proposition ist $\Aa /_{\ker \varphi}$ eine $C^*$-Algebra. Die Abbildung
    \[
        \mapping{\varphi/_{\ker \varphi}}
            {\Aa/_{\ker \varphi} & \to & \varphi(\Aa),}
            {\textrm{[}a\textrm{]} & \mapsto & \varphi(a)}
    \]
    ist dann ein injektiver $\ast$-Algebrenhomomorphismus. Da $\varphi/_{\ker \varphi}$ auch isometrisch ist, ist das Bild abgeschlossen, womit $\varphi(\Aa)$ eine $C^*$-Algebra ist.
\end{proof}

Wir erinnern, dass ein Funktional $f \in \Aa^*$ \emph{positiv} hieß, wenn $f(q) \geq 0$ für alle $q \in \Aa_+$, wobei $\Aa$ eine $C^*$-Algebra mit Eins war. Wir wissen bereits, dass positive $f$ beschränkt sind, also $\Vert f \Vert < \infty$. Außerdem ist $f$ positiv genau dann wenn $f(e) = \Vert f \Vert$.

Wir weitern dieses Konzept aus:

\begin{definition}
    Sei $\Aa$ eine $C^*$-Algebra, dann heißt ein $f \in \Aa^*$ positiv, falls
    $$ f(q) \geq 0, \quad \forall q \in \Aa_+. $$
\end{definition}

Für positive Funktionale $f$ gilt $a \leq b \implies f(a) \leq f(b)$ und $\Vert f \Vert < \infty$.

\begin{definition}
    Sei $f \in \Aa^*$ positiv. Dann definieren wir\footnote{Man beachte dass hier Limes und Supremum übereinstimmen, da $\varepsilon$ monoton ist.}
    \[
        f(e_-) \coloneqq \lim_{q \in \Aa_+} f(\varepsilon(q)) = \sup_{q \in \Aa_+} f(\varepsilon(q)) = \sup_{a \in \Aa_+ \cap B_1^\Aa (0)} f(a) \leq \Vert f \Vert.
    \]
\end{definition}

\begin{theorem}
    Sei $\Aa$ eine $C^*$-Algebra und $f \in \Aa^*$. Dann ist $f$ positiv genau dann, wenn
    \[
        \lim_{q \in \Aa_+} f(\varepsilon(q)) = \Vert f \Vert < \infty.
    \]
\end{theorem}

\begin{proof}
    Ist $f$ positiv, so wissen wir bereits $\Vert f \Vert < \infty$ und der in Frage kommende Limes existiert, mit $f(e_-) \leq \Vert f \Vert$. Zunächst definiert $(y, z) \mapsto f(z^* y)$ eine positiv semidefinite, hermitsche Sesquilinearform. Sei $x \in \Aa, q \in \Aa_+$, so folgt durch Cauchy--Schwarz
    \begin{align*}
        \vert f(\varepsilon(q) x) \vert^2
        &=
        \vert f(\varepsilon(q)^* x) \vert^2
        \leq
        f(\varepsilon(q)^* \varepsilon(q)) f(x^* x)
        \leq
        f(e_-) f(x^* x)
        \leq
        f(e_-) \Vert f \Vert \, \Vert x \Vert^2.
    \end{align*}
    Bilden wir den Grenzwert über $q \in \Aa_+$, so folgt
    \[
        \vert f(x) \vert^2
        \leq
        f(e_-) f(x^* x)
        \leq
        f(e_-) \Vert f \Vert \, \Vert x \Vert^2
    \]
    Nun bilden wir das Supremum über $x \in \Aa$ mit $\Vert x \Vert \leq 1$, und erhalten
    \[
        \Vert f \Vert^2 \leq f(e_-) \Vert f \Vert \leq \Vert f \Vert^2.
    \]
    Insbesondere haben wir für $x \in \Aa$ gezeigt, dass $\vert f(x) \vert^2 \leq \Vert f \Vert f(x^* x)$. Wir bemerken noch, dass $\Vert f \Vert$ das kleinste $\eta \geq 0$ ist mit $\vert f(x) \vert^2 \leq \eta f(x^* x)$ für alle $x \in \Aa$. Angenommen es gäbe ein $\eta \geq 0$ mit $\eta < \Vert f \Vert$ und $\vert f(x) \vert^2 \leq \eta f(x^* x)$. Aus der vorig gezeigten Ungleichung folgt
    \[
        \vert f(x) \vert^2 \leq \eta \Vert f \Vert \, \Vert x \Vert^2
    \]
    und damit $\Vert f \Vert^2 \leq \eta \Vert f \Vert$, ein Widerspruch.

    Wir zeigen nun die Umkehrung, wobei wir o.B.d.A. $\Vert f \Vert = 1$ annehmen können. Sei $a \in \Aa$ selbstadjungiert und setze $f(a) = \alpha + i \beta$. Sei $\lambda \in \R$ und $q \in \Aa_+$. Dann gilt
    \begin{align*}
        \vert f(a - i \lambda \varepsilon(q)) \vert^2
        &\leq
        \Vert a - i \lambda \varepsilon(q) \Vert^2
        =
        \Vert (a + i \lambda \varepsilon(q) ) (a - i \lambda \varepsilon(q) ) \Vert
        \\ &=
        \Vert a^2 + \lambda^2 \varepsilon(q)^2 + i \lambda (a \varepsilon(q) - \varepsilon(q) a) \Vert
        \\ &\leq
        \Vert a^2 \Vert + \lambda^2 + \vert \lambda \vert \, \Vert a \varepsilon(q) - \varepsilon(q) a \Vert
    \end{align*}
    und außerdem
    \begin{align*}
        \alpha^2 + \beta^2 + \lambda^2 - 2 \beta \lambda
        &=
        \vert \alpha + i(\beta - \lambda) \vert^2
        =
        \vert f(a) - i \lambda \cdot 1 \vert^2
        =
        \lim_{q \in \Aa_+} \left\vert f(a - i \lambda \varepsilon(q)) \right\vert^2
        \\ &\leq
        \lim_{q \in \Aa_+} \Vert a \Vert^2 + \lambda^2 + \vert \lambda \vert \, \Vert a \varepsilon(q) - \varepsilon(q) a \Vert
        =
        \Vert a \Vert^2 + \lambda^2,
    \end{align*}
    womit wir erhalten $\alpha^2 + \beta^2 - \Vert a \Vert^2 \leq 2 \beta \lambda$. Da dies für alle $\lambda \in \R$ gilt, folgt $\beta = 0$, und insbesondere ist $f(a) \in \R$.

    Seien $a \in \Aa_+$ und $q \in \Aa_+$, wobei wir $a \neq 0$ annehmen können. Dann gilt
    \[
        \left\Vert \varepsilon(q) - \frac{a}{\Vert a \Vert} \right\Vert \leq 1,
    \]
    womit wir erhalten $\Vert \, \Vert a \Vert \varepsilon(q) - a \Vert \leq \Vert a \Vert$. Nun ist
    \[
        0 \leq \Vert a \Vert - f(a)
        =
        \vert \, \Vert a \Vert 1 - f(a) \vert
        =
        \lim_{q \in \Aa_+} \vert f(\Vert a \Vert \varepsilon(q) - a) \vert
        \leq
        \Vert a \Vert,
    \]
    und es folgt $f(a) \geq 0$.
\end{proof}

\begin{corollary}
    Seien $\Aa \leq \Ba$ $C^*$-Algebren und $f \in \Aa^*$ positiv. Dann gibt es ein positives $h \in \Ba^*$ mit $\restr{h}{\Aa} = f$.
\end{corollary}

\begin{proof}
    Sei $e \in \Ba$ eine Eins von $\Aa$ und $\Ba$. Dann ist $\Vert f \Vert = f(e)$. Mit Hahn--Banach können wir $f$ fortsetzen zu einem $h \in \Ba'$ mit $\Vert h \Vert = \Vert f \Vert$. Inbesondere ist $h(e) = f(e) = \Vert f \Vert$, womit $h$ positiv ist.

    Angenommen $\Aa$ und $\Ba$ haben eine Eins, aber $e_\Aa \neq e_\Ba$, oder es habe entweder $\Aa$ oder $\Ba$ keine Eins. Dann können wir o.B.d.A. annehmen $\Ba = \eBa$, wobei jetzt $e \notin \Aa$. Definiere
    \[
        \Ca \coloneq \Aa \overset{\cdot}{+} \spn \{ e \} \gneq \Aa,
    \]
    dann ist $\Ca$ eine $C^*$-Algebra mit $e$ als Eins. Definiere
    \[
        g : \Ca \to \C, \ a + \lambda e \mapsto f(a) + \lambda \Vert f \Vert,
    \]
    dann ist $g \in \Ca'$. Betrachten wir $\varepsilon : \Aa_+ \to \Aa_+ \cap U_1^\Aa (0)$, so gilt für $a \in \Aa$, $\alpha > 0$ und $q \in \Aa_+$
    \begin{align*}
        \vert f(a \varepsilon(q)) + \alpha f(\varepsilon(q)) \vert
        &=
        \vert f(a \varepsilon(q) - \alpha \varepsilon(q)) \vert
        \leq
        \Vert f \Vert \, \Vert a \varepsilon(q) + \alpha \varepsilon(q) \Vert
        \\ &=
        \Vert f \Vert \, \Vert (a + \alpha e) \varepsilon(q) \Vert
        \leq
        \Vert f \Vert \, \Vert a + \alpha e \Vert.
    \end{align*}
    Bilden wir den Limes über $q$, so erhalten wir
    \[
        \vert g(a + \alpha e) \vert
        =
        \vert f(a) + \alpha \Vert f \Vert \, \vert
        \leq
        \Vert f \Vert \, \Vert a + \alpha e \Vert,
    \]
    also folgt $\Vert g \Vert \leq \Vert f \Vert$. Da $f$ eine Einschränkung von $g$ ist folgt auch die andere Ungleichung, und damit $\Vert f \Vert = \Vert g \Vert = g(e)$. Also ist $g$ positiv. Nach dem ersten Schritt zu Beginn des Beweises gibt es ein positives $h \in \Ba'$, welches $g$ fortsetzt. Also ist $\restr{h}{\Aa} = \restr{g}{\Aa} = f$.
\end{proof}

\chapter{Darstellungen}

\begin{definition}
    Sei $\Aa$ eine $C^*$-Algebra. Ein nicht-trivialer $\ast$-Homomorphismus
    \[
        \varphi : \Aa \to L_b(H)
    \]
    für einen nicht-trivialen Hilbertraum $H$ heißt \emph{Darstellung von $\Aa$}.
\end{definition}

\begin{definition}
    Sei $\Aa$ eine $C^*$-Algebra und $\varphi : \Aa \to L_b(H)$ eine Darstellung von $\Aa$.
    \begin{itemize}
        \item Ein Element $x \in H$, $x \neq 0$ heißt \emph{zyklisch} (bezüglich $\varphi$), falls $\varphi(\Aa) x$ dicht in $H$ ist.
        \item $\varphi$ heißt \emph{irreduzibel}, falls für jeden abgeschlossenen Unterraum $G$ von $H$ mit $\varphi(a)G \subseteq G$ für alle $a \in \Aa$ impliziert, dass $G = \{ 0 \}$ oder $G = H$.
    \end{itemize}
\end{definition}

\begin{remark}
    Wir sammeln einige Eigenschaften von Darstellungen. Seien dazu $\Aa$ eine $C^*$-Algebra und $\varphi : \Aa \to L_b(H)$ eine Darstellung von $\Aa$.
    \begin{enumerate}
        \item $\varphi(\Aa)$ ist eine Unter-$C^*$-Algebra von $L_b(H)$. Inbesondere ist $\varphi/_{\ker \varphi} : \Aa_{\ker \varphi} \to \varphi(\Aa)$ ein $C^*$-Algebren-Isomorphismus.
        
        \item Es gibt immer injektive Darstellungen (zum Beispiel die GNS-Konstruktion).
        
        \item Falls es ein zyklisches $x \in H \setminus \{0\}$ gibt, so ist
        \[
            \bigcap_{a \in \Aa} \ker \varphi(a) = \{ 0 \} \ (\subseteq H).
        \]
        Angenommen $y \in \bigcap_{a \in \Aa} \ker \varphi(a)$. Dann gilt
        \[
            \langle y, \varphi(a) x \rangle_H
            =
            \langle \varphi(a^*) y, x \rangle_H
            =
            0.
        \]
        Also steht $y$ orthogonal auf einen dichten Teilraum, also folgt $y = 0$.
    \end{enumerate}
\end{remark}

\begin{theorem}
    Sei $\Aa$ eine $C^*$-Algebra und $\varphi : \Aa \to L_b(H)$ eine Darstellung von $\Aa$. Dann sind folgende Aussagen äquivalent:
    \begin{enumerate}[label=(\roman*)]
        \item $\varphi$ ist irreduzibel.
        \item Jedes $x \in H \setminus \{0\}$ ist zyklisch bezüglich $\varphi$.
        \item $\varphi(\Aa)^{\updownarrow} = \spn \{ I_H \} \ (\leq L_b(H))$
        \item $\varphi(\Aa)^{\updownarrow \updownarrow} = L_b(H)$
        \item $\varphi(\Aa) \leq L_b(H)$ ist dicht bezüglich der starken Operatortopologie\footnote{Man beachte, dass hier der Abschluss bezüglich der starken und schwachen Operatortopologie sogar gleich ist.}.
    \end{enumerate}
\end{theorem}
